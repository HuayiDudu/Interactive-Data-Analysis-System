# 交互式数据分析系统 - 开发说明文档

---

## 一、项目介绍

本系统实现一个Web版数据分析工具，支持上传CSV/Excel文件，进行数据清洗（缺失值处理、异常值检测）、可视化（至少3种图表）、机器学习分析（K-Means聚类或线性回归），并导出结果。采用分层架构，MVP阶段使用临时文件存储，扩展阶段可增加SQLite、多用户登录等。

---

## 三、分层最小完成内容（MVP）与扩展要求

### 0. 核心架构抽象（贯穿各层的通用组件）

为消除隐式状态、统一数据访问接口并降低耦合，系统定义以下抽象，所有层级必须严格遵循。

#### 0.1 数据集引用（值对象）

```python
# value_objects.py
from dataclasses import dataclass

@dataclass(frozen=True)
class DatasetRef:
    """唯一标识一个已保存的数据集，由数据访问层生成，上层透传。"""
    id: str  # MVP中为内部映射的UUID，扩展后为数据库自增ID的字符串形式
```

#### 0.2 数据仓库抽象基类

```python
# repositories/base.py
from abc import ABC, abstractmethod
import pandas as pd
from value_objects import DatasetRef

class DataRepository(ABC):
    """数据访问层统一接口，所有Service必须依赖此抽象，而非具体实现。"""

    @abstractmethod
    def save_data(self, df: pd.DataFrame, context: dict | None = None) -> DatasetRef:
        """保存DataFrame，返回唯一引用。context可携带user_id等信息，MVP可忽略。"""
        ...

    @abstractmethod
    def load_data(self, ref: DatasetRef) -> pd.DataFrame:
        """根据引用加载DataFrame，若无效则抛出ValueError。"""
        ...

    @abstractmethod
    def delete_data(self, ref: DatasetRef) -> None:
        """删除引用对应的数据集，用于清理或回滚。"""
        ...

    # 扩展接口（MVP可选实现，默认抛出NotImplementedError）
    def list_datasets(self, user_id: int, page: int, per_page: int,
                      search: str = "", order_by: str = "created_at",
                      order_dir: str = "desc") -> dict:
        raise NotImplementedError
```

#### 0.3 数据集存储格式

- **MVP**：内部可使用CSV临时文件，由`FileRepository`管理。
- **扩展**：必须使用**Apache Parquet**格式（`df.to_parquet(engine='pyarrow')`）存储于SQLite的BLOB字段，避免pickle的安全风险和版本兼容问题。

### 4. 数据访问层（Repository）

#### MVP实现：文件仓库（`FileRepository`）

基于临时文件系统，通过**应用级字典维护`DatasetRef`到文件路径的映射**，完全不依赖`flask.session`。

```python
# repositories/file_repo.py
import os
import uuid
import pandas as pd
from repositories.base import DataRepository
from value_objects import DatasetRef
from config import UPLOAD_FOLDER

class FileRepository(DataRepository):
    """MVP文件数据仓库，内部使用字典维护引用到文件路径的映射。"""
    
    def __init__(self):
        self._ref_map: dict[str, str] = {}   # ref_id -> 绝对文件路径

    def save_data(self, df: pd.DataFrame, context: dict | None = None) -> DatasetRef:
        """保存DataFrame为CSV临时文件，返回新生成的DatasetRef。"""
        ref_id = uuid.uuid4().hex
        filepath = os.path.join(UPLOAD_FOLDER, f"{ref_id}.csv")
        df.to_csv(filepath, index=False)
        self._ref_map[ref_id] = filepath
        return DatasetRef(ref_id)

    def load_data(self, ref: DatasetRef) -> pd.DataFrame:
        """根据引用加载DataFrame，若无效则抛出ValueError。"""
        filepath = self._ref_map.get(ref.id)
        if not filepath or not os.path.exists(filepath):
            raise ValueError(f"数据集不存在或已被清理: {ref.id}")
        return pd.read_csv(filepath)

    def delete_data(self, ref: DatasetRef) -> None:
        """删除引用对应的临时文件及映射。"""
        filepath = self._ref_map.pop(ref.id, None)
        if filepath and os.path.exists(filepath):
            os.remove(filepath)
```

**说明**：
- 上传后返回的`DatasetRef.id`会由控制层返回给前端，后续请求均需携带此id。
- 映射字典存储在内存中，重启后清空，符合MVP临时性特征。正式环境可换为Redis等。
- 清洗等操作会创建新的CSV文件并返回新的`DatasetRef`，原文件不受影响，实现不可变数据操作。

#### 扩展要求：SQLite仓库（`SQLiteRepository`）

实现`DataRepository`接口，使用SQLite存储，数据集以**Parquet二进制**存入BLOB。

```python
# repositories/sqlite_repo.py
import sqlite3
import pandas as pd
import pyarrow as pa
import pyarrow.parquet as pq
from io import BytesIO
from repositories.base import DataRepository
from value_objects import DatasetRef

class SQLiteRepository(DataRepository):
    def __init__(self, db_path: str = "analysis.db"):
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        # ... 创建表结构（见数据存储层）

    def save_data(self, df: pd.DataFrame, context: dict | None = None) -> DatasetRef:
        """将DataFrame序列化为Parquet字节流存入datasets表。"""
        user_id = context.get('user_id') if context else None
        name = context.get('name', 'unnamed')
        table = pa.Table.from_pandas(df)
        sink = BytesIO()
        pq.write_table(table, sink)
        data_bytes = sink.getvalue()
        with sqlite3.connect(self.db_path) as conn:
            cursor = conn.execute(
                "INSERT INTO datasets (user_id, name, data_parquet) VALUES (?, ?, ?)",
                (user_id, name, data_bytes)
            )
            return DatasetRef(str(cursor.lastrowid))

    def load_data(self, ref: DatasetRef) -> pd.DataFrame:
        """从数据库加载Parquet数据并还原为DataFrame。"""
        with sqlite3.connect(self.db_path) as conn:
            row = conn.execute(
                "SELECT data_parquet FROM datasets WHERE id = ?", (int(ref.id),)
            ).fetchone()
            if not row:
                raise ValueError(f"数据集不存在: {ref.id}")
            buf = BytesIO(row[0])
            return pq.read_table(buf).to_pandas()

    def delete_data(self, ref: DatasetRef) -> None:
        with sqlite3.connect(self.db_path) as conn:
            conn.execute("DELETE FROM datasets WHERE id = ?", (int(ref.id),))

    def list_datasets(self, user_id: int, page: int, per_page: int,
                      search: str, order_by: str, order_dir: str) -> dict:
```

### 5. 数据存储层

**MVP**：操作系统临时文件（如 `/tmp/<ref_id>.csv`）。由`FileRepository`自动管理，路径对外不可见。  
**扩展**：SQLite数据库文件 `analysis.db`。表结构设计如下（使用Parquet格式存储数据集）：

```sql
-- 用户表（多用户登录）
CREATE TABLE users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 数据集表（存储每次保存的数据集快照，使用Parquet二进制）
CREATE TABLE datasets (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    name TEXT NOT NULL,                -- 用户自定义名称或原始文件名
    data_parquet BLOB NOT NULL,        -- Parquet序列化字节（pyarrow序列化）
    description TEXT,                  -- 可选描述
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
CREATE INDEX idx_datasets_user_id ON datasets(user_id);
CREATE INDEX idx_datasets_created_at ON datasets(created_at);

-- 清洗规则表（自动化清洗规则配置）
CREATE TABLE cleaning_rules (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    rule_name TEXT NOT NULL,
    rule_config TEXT NOT NULL,         -- JSON字符串，存储清洗策略
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
CREATE INDEX idx_cleaning_rules_user_id ON cleaning_rules(user_id);

-- 分析历史记录表（可选，用于追踪用户分析行为）
CREATE TABLE analysis_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    dataset_id INTEGER NOT NULL,
    algorithm TEXT NOT NULL,           -- 'kmeans' 或 'linear_regression'
    parameters TEXT,                   -- JSON参数
    result_summary TEXT,               -- JSON结果摘要（如惯性、R²等）
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (dataset_id) REFERENCES datasets(id)
);
```

**层间依赖规则（必须严格遵守）**

1. 可以调用：操作系统API、文件系统、数据库引擎、Parquet序列化库（pyarrow）
2. 禁止调用：任何上层代码，业务逻辑、Web框架等

### 4.3 业务逻辑层 ↔ 数据访问层

所有交互通过`DataRepository`抽象接口。Service层方法示例：

## 五、验收标准

### MVP必做项

| 模块     | 验收点       | 通过标准                                                     |
| -------- | ------------ | ------------------------------------------------------------ |
| 数据管理 | 文件上传     | 支持.csv和.xlsx文件，上传后显示列名、前5行、总行列数，并返回可用数据集标识 |
| 数据管理 | 导出         | 可导出指定数据集为CSV或Excel，文件内容与系统内一致           |
| 数据清洗 | 缺失值处理   | 支持均值填充、中位数填充、删除行三种策略，操作后预览更新，生成新数据集 |
| 数据清洗 | 异常值检测   | 使用IQR方法检测并报告异常值数量，可删除异常值所在行          |
| 可视化   | 生成3种图表  | 用户可自由选择X/Y列，生成散点图、折线图、柱状图              |
| 分析功能 | 实现一种分析 | 根据自选算法（K-Means或线性回归）提供可调参数，展示结果      |
| Web界面  | 完整流程     | 无需刷新页面即可完成 上传→清洗→可视化→分析→导出，且每次操作传递正确的dataset_id |
| Web界面  | 响应式布局   | 在常见PC分辨率下布局合理，控件正常显示                       |
